## Sign up as a Spikex.com user

Create API Key
Please refer to [Create API Key](https://www.spikex.com/en/accounts/user/api)

Import Perp API 


from pyspikex.perp import Perp
Output:
## Fill in your API key and Secret key 

api_key = "xxxxx"
secret_key = "xxxxx"
spikex = Perp(host="https://fapi.spikex.com", access_key=api_key, secret_key=secret_key)
## Get available funds 

print(spikex.get_account_capital())
## Get market data book tickers 

print(spikex.get_book_ticker(symbol='btc_usdt'))
## Get market data depth 

print(spikex.get_depth(symbol='btc_usdt', depth=10))
## Get market data klines 

print(spikex.get_depth(symbol='btc_usdt', depth=10))
## Get symbol information 

print(spikex.get_market_config(symbol='btc_usdt'))
## Place a limit order 

res = spikex.send_order(symbol='btc_usdt', price=10000, amount=1, order_side='BUY', order_type='LIMIT', position_side='LONG')
print(res)
## Place a market order 

res = spikex.send_order(symbol='btc_usdt', price=10000, amount=1, order_side='BUY', order_type='MARKET', position_side='LONG')
print(res)
## Cancel order 

res = spikex.cancel_order(order_id=12345678)
print(res)
