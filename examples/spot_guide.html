## Sign up as a Spikex.com user
Please refer to [Create account](https://www.spikex.com/en/accounts/register)

Create API Key
Please refer to [Create API Key](https://www.spikex.com/en/accounts/user/api)

Import Spot API 


from pyspikex.spot import Spot
## Fill in your API key and Secret key 

api_key = "xxxxx"
secret_key = "xxxxx"
spikex = Spot(host="https://sapi.spikex.com", access_key=api_key, secret_key=secret_key)
## Get available funds 

print(spikex.balance("usdt"))
## Get market data tickers 

print(spikex.get_tickers(symbol='btc_usdt'))
## Get market data depth 

print(spikex.get_depth(symbol='btc_usdt'))
## Get market data book tickers 

print(spikex.get_tickers_book(symbol='btc_usdt'))
## Get market data klines 

print(spikex.get_kline(symbol='btc_usdt', interval="1m"))
## Get symbol information 

print(spikex.get_symbol_config(symbol='btc_usdt'))
## Place a limit order 

res = spikex.order(symbol='btc_usdt', price=10000, quantity=0.001, side='BUY', type='LIMIT')
print(res)
## Place a market order by quote_qty 

res = spikex.order(symbol='smh_usdt', time_in_force="IOC", quote_qty=2, side='BUY', type='MARKET')
print(res)
## Place a market order by quantity 

res = spikex.order(symbol='smh_usdt', time_in_force="IOC", quantity=2, side='BUY', type='MARKET')
print(res)
## Cancel order 

res = spikex.cancel_order(order_id=12345678)
print(res)
